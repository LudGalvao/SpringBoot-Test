(Sistema de armazenamento digital de objetos em bancos - desejo ver que objetos foram penhorados e o valor da penhora por agencia)

CREATE DATABASE agencia;

-- Correção da criação da tabela de agências
CREATE TABLE agencias (
    id INT UNSIGNED PRIMARY KEY AUTO_INCREMENT,
    nome_agencia VARCHAR(100) NOT NULL,
    endereco VARCHAR(255),
    cidade VARCHAR(100),
    estado VARCHAR(50)
);

-- Correção da criação da tabela de objetos penhorados
CREATE TABLE objetos_penhorados (
    id INT PRIMARY KEY AUTO_INCREMENT,
    nome_objeto VARCHAR(100) NOT NULL,
    descricao TEXT,
    valor_penhora DECIMAL(10, 2),
    agencia_id INT UNSIGNED,
    FOREIGN KEY (agencia_id) REFERENCES agencias(id)
);
